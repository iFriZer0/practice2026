	\chapter{Назначение программы}
	\label{chapter_purpose}
	
	ПО предназначено для работы с драйверами следующих устройств:
	
	\begin{itemize}
		\item плата RS-485;
		\item плата МКО;
		\item плата РК-ПКУ.
	\end{itemize}
	
	ПО решает следующие задачи:
	
	\begin{itemize}
		\item вызов всех команд, предусмотренных драйверами;
		\item обработка и отображение принятой информации;
		\item формирование передаваемой информации как с ошибками, так и без ошибок;
		\item обеспечение перекладки выхода одного драйвера на вход другого;
		\item логирование работы.
	\end{itemize}

	\chapter{Условия выполнения ПО}
	\label{chapter_conditions}
	
	\section{Аппаратные требования}
	
	Для работы ПО достаточно выполнения следующих условий:
	
	\begin{itemize}
		\item архитектура процессора --- x86-64;
		\item объём оперативной памяти не менее $512\ \text{МБ}$;
		\item объём внешней памяти не менее $256\ \text{МБ}$.
	\end{itemize}

	\section{Программные требования}
	
	ПО работает в операционной системе Linux на дистрибутиве, совместимом с Ubuntu 22.04.5LTS. Для выполнения требуются права на чтение и запись файлов в директории с ПО.
	
	Необходим установленный интерпретатор языка программирования Python версии не ниже 3.10.12. Путь к интерпретатору должен быть указан в переменной окружения $\mathtt{PATH}$.
	
	\chapter{Выполнение программы}
	\label{chapter_process}
	
	\section{Загрузка программы}
	
	Для выполнения программы должны быть загружены:
	
	\begin{itemize}
		\item исполняемый файл клиента \texttt{gui/src/build/out/ui};
		\item исполняемый файл микросервиса МКО \texttt{mko/bin/mko-service};
		\item конфигурационные файлы микросервиса МКО\linebreak\texttt{mko/config/config.yaml} и \texttt{mko/config/local.env};
		\item драйвер МКО;
		\item исходный код микросервиса РК-ПКУ в директории \texttt{pku/src/service};
		\item скрипт запуска микросервиса РК-ПКУ \texttt{pku/src/service/run.sh};
		\item конфигурационные файлы микросервиса РК-ПКУ\linebreak\texttt{pku/src/configuration/driver\_address.json} и\linebreak\texttt{pku/src/configuration/pku\_service\_address.txt};
		\item драйвер РК-ПКУ;
		\item исполняемый файл микросервиса RS-485\linebreak\texttt{rs/build/rs485\_microservice};
		\item конфигурационный файл микросервиса RS-485 \texttt{rs/config/config.yaml};
		\item драйвер RS-485.
	\end{itemize}

	\section{Запуск программы}

	Для запуска программы достаточно выполнить следующие действия в порядке их перечисления:
	
	\begin{itemize}
		\item запустить драйвер МКО;
		\item запустить драйвер РК-ПКУ;
		\item запустить драйвер RS-485;
		\item запустить микросервис МКО с помощью исполняемого файла\linebreak\texttt{mko/bin/mko-service} из директории \texttt{mko};
		\item запустить микросервис РК-ПКУ с помощью скрипта\linebreak\texttt{pku/src/service/run.sh} из любой директории;
		\item запустить микросервис RS-485 с помощью исполняемого файла\linebreak\texttt{rs/build/rs485\_microservice} из директории \texttt{rs};
		\item запустить клиент с помощью исполняемого файла\linebreak\texttt{gui/src/build/out/ui} из директории \texttt{gui/src}.
	\end{itemize}
